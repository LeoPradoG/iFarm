--TABELA MYSQL PARA RECEBER OS DADOS DAS BATELADAS 

CREATE TABLE fluxo_balanca(

CD_REGISTRO 		INT AUTO_INCREMENT PRIMARY KEY,
DT_BATELADA 		varchar(50) ,
HORA        		varchar(50),
QTD_BATELADA 		int(11) ,
TEMP_ENCH		float,
TEMP_DESC		int(11),
CARGA_INICIAL		int(11),
CARGA_FINAL		float,
CARGA_RESIDUAL		float,
CARGA_ENTREGUE		float,
CARGA_TOTAL_ENTREGUE 	float,
ACUMULADOR_GERAL 	float,
VAZAO_INSTAT 		int(11)
);

--TABELA PARA CRIAÇÃO DE USUARIOS ( ACESSOS ) 
CREATE TABLE usuarios (
CD_USUARIO 		INT AUTO_INCREMENT PRIMARY KEY,
DS_LOGIN 		varchar(100),
DS_SENHA		varchar(200),
SN_PRIMEIRO_ACESSO 	varchar(1),
SN_ADMINISTRADOR	varchar(1),
CD_TIPO_ACESSO		INT,
NM_USUARIO		varchar(100),
TP_SEXO			varchar(1),
DT_NASCIMENTO		date,
CD_USUARIO_CADASTRO 	varchar(100),
HR_CADASTRADO		DATETIME,
CD_USUARIO_ULT_ALT	varchar(100),
HR_ULT_ALT		DATETIME

);
